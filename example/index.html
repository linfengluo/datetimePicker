点我选择时间
你选择的时间：
    
new DateTimePicker({
    id: 'timeSelect',
    type: 'time',
    onSelect(val, date) {
        document.querySelector('#timeSelectValue span').innerText = date
    }
})
点我选择日期
你选择的日期：
    
new DateTimePicker({
  id: 'dateSelect',
  type: 'date',
  format: 'YYYY/MM/DD',
  onSelect(val, date) {
    document.querySelector('#dateSelectValue span').innerText = date
  }
})
点我选择日期和时间
你选择的日期和时间：
    
new DateTimePicker({
  id: 'dateTimeSelect',
  type: 'dateTime',
  onSelect(val, date) {
    document.querySelector('#dateTimeSelectValue span').innerText = date
  }
})
